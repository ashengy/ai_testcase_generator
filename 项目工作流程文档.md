## 项目概述

这是一个基于 **PyQt5** 和 **DeepSeek AI** 的智能化测试用例生成工具，支持从需求文档（Word/PDF）中自动提取内容，通过 AI 分析生成结构化的测试用例。

### 核心功能
- ✅ 支持 Word (.docx) 和 PDF 格式需求文档
- ✅ AI 图片分析与文字识别（可选）
- ✅ 多种测试用例设计方法（正交分析、边界值、等价类等）
- ✅ 异步 AI 推理，实时显示进度
- ✅ 智能 JSON 修复与格式化
- ✅ 多格式导出（XLSX/Word/Txt/MD/JSON）

```python
def load_directory(self, directory):
    try:
        # 1. 保存当前目录
        self.current_dir = directory
        
        # 2. 清空文件列表
        self.file_list.clear()
        
        # 3. 验证目录存在性
        if not os.path.exists(directory):
            QMessageBox.warning(self, "路径错误", f"目录不存在：{directory}")
            return
        
        # 4. 扫描有效文件 (.docx, .pdf)
        valid_extensions = ('.docx', '.pdf')
        for fname in sorted(os.listdir(directory)):
            full_path = os.path.join(directory, fname)
            if os.path.isfile(full_path) and fname.lower().endswith(valid_extensions):
                # 5. 创建列表项
                item = QListWidgetItem(fname)
                item.setData(Qt.UserRole, full_path)  # 存储完整路径
                item.setToolTip(full_path)
                self.file_list.addItem(item)
        
        # 6. 显示文件数量
        self.statusBar().showMessage(f"已加载 {self.file_list.count()} 个文档", 3000)
    
    except Exception as e:
        QMessageBox.critical(self, "加载错误", str(e))
```

```python
def extract_pdf_text_with_image_list(pdf_path, image_replacement_list):
    """
    提取 PDF 文本和图片位置信息
    """
    with open(pdf_path, 'rb') as f:
        reader = PdfReader(f)
        pages_text = []
        
        for page in reader.pages:
            text = page.extract_text()
            pages_text.append(text if text else "")
        
        return '\n'.join(pages_text)
```

```python
def run(self):
    try:
        # 1. 判断文件类型
        file_type = os.path.splitext(self.file_path)[1]
        
        # 2. 根据开关决定是否分析
        if self.analyzer_enable:
            self.current_status.emit("----开始启动 AI 分析图片...----\n")
            
            # 3. 根据文件类型选择分析器
            if file_type == ".pdf":
                # PDF 图片分析
                analyzer = PDFImageAIAnalyzer(
                    api_key=self.image_api_key,
                    model_name="qwen-vl-plus"
                )
                replacements = analyzer.process_pdf_images(
                    self.file_path,
                    batch_delay=self.batch_delay
                )
            
            elif file_type == ".docx":
                # Word 图片分析
                analyzer = WordImageAIAnalyzer(
                    api_key=self.image_api_key,
                    model_name="qwen-vl-plus"
                )
                replacements = analyzer.process_word_images(
                    self.file_path,
                    batch_delay=self.batch_delay
                )
            
            # 4. 显示每张图的分析结果
            for i, v in enumerate(replacements):
                if v == "":
                    v = "无效图片，已过滤"
                self.current_status.emit(f"第{i + 1}张图分析结果:\n {v} \n")
            
            # 5. 提取文档文本（包含图片描述）
            if file_type == ".pdf":
                context = extract_pdf_text_with_image_list(
                    self.file_path,
                    image_replacement_list=replacements
                )
            elif file_type == ".docx":
                context = insert_image_position_with_list(
                    self.file_path,
                    image_replacement_list=replacements
                )
            
            # 6. 清洗文本
            context = context.replace("◦", "")
            context = context.replace(" ", "")
            context = context.replace("\n\n", "\n")
            
            # 7. 发射完成信号
            self.finished.emit(context)
        
        except Exception as e:
            self.error.emit(f"ImageAnalyzer 运行异常：{e}")
```

### 六、AI 推理线程详细流程

#### 📍 `core/worker.py::GenerateThread.run()`

```python
def run(self):
    self.current_status.emit("----开始生成测试用例...----\n")
    
    try:
        all_result_str = ""
        count = len(self.context)
        
        # 1. 遍历所有文本块
        for n, context_chunk in enumerate(self.context):
            # 同步进度
            self.current_stage.emit(f"当前推理进度：{n}/{count}")
            
            # 2. 调用 AI 生成
            result = self.generate_cases(context_chunk)
            
            if result is not None and isinstance(result, str):
                all_result_str += result
                print(f"第 {n + 1} 个 chunk 结果长度：{len(result)}")
            
            # 同步完成进度
            self.current_stage.emit(f"当前推理进度：{n + 1}/{count}")
        
        # 3. 尝试解析 JSON
        if all_result_str and ('json' in all_result_str.lower() or 
                               '{' in all_result_str or '[' in all_result_str):
            # 4. 提取和修复 JSON
            extracted_result = self.extract_json_objects(all_result_str)
            
            if extracted_result:
                # 5. 格式化测试用例
                all_result_str = self.reformat_test_cases(extracted_result)
        
        # 6. 发射完成信号
        self.finished.emit(all_result_str)
        self.current_status.emit("----本轮已执行完成！----")
    
    except Exception as e:
        self.error.emit(str(e))
```

#### 📍 `core/worker.py::GenerateThread.generate_cases()`

```python
def generate_cases(self, chunk_data):
    # 1. 初始化客户端
    client = OpenAI(
        api_key=self.api_key if self.api_key else 'sk-8509fd7dfb9248e49334111e24141d22',
        base_url='https://api.deepseek.com'
    )
    
    reasoning_content = ""  # 思考过程
    answer_content = ""     # 最终回复
    is_answering = False    # 是否结束思考开始回复
    
    # 2. 创建聊天请求（流式）
    completion = client.chat.completions.create(
        model="deepseek-reasoner",
        messages=[{
            'role': 'user',
            'content': f'''
            所在行业：{self.job_area}；
            文档内容：{chunk_data}；
            生成的用例类型：{self.func_type}；
            用例设计方法：{self.design_method}；
            提示词：{self.prompt}；
            '''
        }],
        stream=True,
        max_tokens=64000 - 4000
    )
    
    # 3. 流式处理响应
    for chunk in completion:
        delta = chunk.choices[0].delta
        
        # 接收思考过程
        if hasattr(delta, 'reasoning_content') and delta.reasoning_content:
            reasoning_content += delta.reasoning_content
        else:
            # 开始回复
            if delta.content != "" and not is_answering:
                is_answering = True
            
            # 接收回复内容
            answer_content += delta.content
            
            # 实时显示到界面
            self.current_status.emit(f"\n{answer_content}\n")
    
    return answer_content
```

```python
def extract_json_objects(self, text: str) -> List[Dict]:
    """
    从文本中提取 JSON 对象并返回列表
    """
    # 1. 调用修复函数
    result = self.fix_and_extract_json(text)
    
    # 2. 确保返回的是列表
    if isinstance(result, list):
        filtered_list = []
        for i, item in enumerate(result):
            if isinstance(item, dict):
                filtered_list.append(item)
            elif isinstance(item, list):
                filtered_list.extend(self._merge_json_objects([item]))
        
        return filtered_list
    
    elif isinstance(result, dict):
        return [result]
    
    else:
        return []
```

#### 📍 `core/worker.py::GenerateThread.fix_and_extract_json()`

```python
def fix_and_extract_json(self, text: str) -> Union[List[Dict], Dict, str]:
    """
    修复 JSON 文本的完整流程
    """
    # 步骤 1: 清理文本
    cleaned_text = self._clean_json_text(text)
    # 移除 ```json 和 ``` 标记
    # 移除多余空行
    
    # 步骤 2: 尝试直接解析
    try:
        result = json.loads(cleaned_text)
        return result
    except json.JSONDecodeError as e:
        print(f"✗ JSON 解析错误：{e}")
    
    # 步骤 3: 修复结构
    fixed_json = self._repair_json_structure(cleaned_text)
    # 补全括号
    # 移除末尾逗号
```

```python
def _repair_json_structure(self, text: str) -> str:
    """修复 JSON 结构问题"""
    text = text.strip()
    
    # 1. 确保以 { 或 [ 开头
    if not text.startswith(('{', '[')):
        brace_pos = text.find('{')
        bracket_pos = text.find('[')
        
        if brace_pos != -1:
            text = text[brace_pos:]
        elif bracket_pos != -1:
            text = text[bracket_pos:]
        else:
            return f'[{text}]'
    
    # 2. 检查括号匹配
    stack = []
    for i, char in enumerate(text):
        if char in '([{':
            stack.append((char, i))
        elif char in ')]}':
            if stack and stack[-1][0] == char_map[char]:
                stack.pop()
    
    # 3. 补全缺失的括号
    while stack:
        missing_char, pos = stack.pop()
        if missing_char == '{':
            text += '}'
        elif missing_char == '[':
            text += ']'
    
    # 4. 移除闭合括号前的逗号
    text = re.sub(r',(\s*[}\]])', r'\1', text)
    
    # 5. 修复连续逗号
    text = re.sub(r',\s*,', ',', text)
    
    return text
```

```python
def normalize_data(data):
    """
    统一数据格式，将多层嵌套转换为单层列表
    输入：[{key:[{dict1},{dict2}]}]
    输出：[{dict1},{dict2}]
    """
    if not isinstance(data, list) or len(data) == 0:
        return []
    
    first_item = data[0]
    
    # 检查是否为 data2 格式
    if (isinstance(first_item, dict) and
        isinstance(next(iter(first_item.values())), list)):
        return next(iter(first_item.values()))
    
    return data
```

### 九、文本分块算法

#### 📍 `core/utils.py::chunk_text()`

```python
def chunk_text(text, chunk_size=8000, overlap=500):
    """
    滑动窗口分块算法
    """
    if text:
        # 1. 清理符号
        clear_text = text.replace("·", "")
        
        chunks = []
        start = 0
        text_length = len(clear_text)
        
        # 2. 滑动窗口切分
        while start < text_length:
            end = min(start + chunk_size, text_length)
            chunk = clear_text[start:end]
            chunks.append(chunk)
            
            # 3. 滑动：移动 chunk_size - overlap
            start += chunk_size - overlap
        
        return chunks
```

## 关键算法说明

### 1. 滑动窗口分块算法

**目的：** 将长文本分割成适合 AI 处理的小块，同时保持上下文连贯性

**参数：**
- `chunk_size`: 每块最大 8000 字符
- `overlap`: 相邻块重叠 500 字符

**优势：**
- 避免超出 AI 模型的 token 限制
- 重叠部分保证上下文连续性
- 防止重要信息被切断

### 2. JSON 容错解析算法

**多级修复策略：**

1. **直接解析** - 如果文本已经是有效 JSON
2. **结构修复** - 补全括号、移除多余逗号
3. **对象提取** - 从文本中提取所有 JSON 片段
4. **合并结果** - 将多个对象合并为统一格式

**修复的问题类型：**
- 缺失的闭合括号
- 末尾多余的逗号
- markdown 代码块标记
- 未转义的引号

### 3. 数据标准化算法

**目的：** 统一不同的数据结构，便于后续处理

**输入格式：**
```python
[{key: [{dict1}, {dict2}]}]  # 嵌套结构
```

**输出格式：**
```python
[{dict1}, {dict2}]  # 扁平化结构
```
